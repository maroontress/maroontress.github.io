    "update_example.db", m => Console.WriteLine(m()));

{
    ["$id"] = 2L,
    ["$newState"] = "Closed",
};

{
    q.Update<Issue>("i")
        .Set("state = $newState")
        .Where("i.id = $id")
        .Execute(map);
});
